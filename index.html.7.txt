    function showQuestion() {
        const item = questions[currentIdx];
        document.getElementById("progress").innerText = `Câu ${currentIdx + 1}/15`;
        document.getElementById("q-text").innerText = item.q;
        document.getElementById("options").innerHTML = item.a.map((opt, i) =>

        document.getElementById("quiz-section").style.display = "block";
        document.getElementById("feedback").style.display = "none";
    }

        const isCorrect = (idx === questions[currentIdx].c);
        document.getElementById("quiz-section").style.display = "none";
        document.getElementById("feedback").style.display = "block";

            document.getElementById("fb-title").innerHTML = "<span style='color:green'>10 ĐIỂM GIỎI QUÁ!</span>";

        currentIdx++;

        else {
            alert("Chúc mừng bạn đã hoàn thành bài thi từ " + TEN_TIEN_BOI);
            currentIdx = 0; showQuestion();
        }
